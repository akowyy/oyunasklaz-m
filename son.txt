    <script>
        const canvas = document.getElementById('game');
        const ctx = canvas.getContext('2d');
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;

        let groundOffset = 0;

        function draw() {
            // Gökyüzü (Gün batımı rengi)
            let grad = ctx.createLinearGradient(0, 0, 0, canvas.height);
            grad.addColorStop(0, '#ff9a9e');
            grad.addColorStop(1, '#fad0c4');
            ctx.fillStyle = grad;
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // Kumsal (İlerleyen yol efekti)
            ctx.fillStyle = '#e6c2a8';
            ctx.fillRect(0, canvas.height * 0.7, canvas.width, canvas.height * 0.3);

            // Karakterler (İki kişi - Basit silüetler)
            ctx.fillStyle = '#5d4037';
            let yPos = canvas.height * 0.75 + Math.sin(Date.now() / 200) * 10;
            // Kişi 1
            ctx.beginPath(); ctx.arc(canvas.width / 2 - 30, yPos, 15, 0, Math.PI * 2); ctx.fill();
            // Kişi 2
            ctx.beginPath(); ctx.arc(canvas.width / 2 + 30, yPos, 15, 0, Math.PI * 2); ctx.fill();

            requestAnimationFrame(draw);
        }
        draw();
    </script>
